Learn new model with gridsearch and train-test set

AAA

# Import library
import clustimage

# Load example data set
X,y_true = clustimage.load_example()

# Retrieve URLs of malicous and normal urls:
model = clustimage.fit_transform(X, y_true, pos_label='bad', train_test=True, gridsearch=True)

# The test error will be shown
results = clustimage.plot(model)

# Import library
import clustimage

# Load example data set
X,y_true = clustimage.load_example()

# Retrieve URLs of malicous and normal urls:
model = clustimage.fit_transform(X, y_true, pos_label='bad', train_test=False, gridsearch=True)

# The train error will be shown. Such results are heavily biased as the model also learned on this set of data
results = clustimage.plot(model)
